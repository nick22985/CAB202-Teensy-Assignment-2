void zombe_physics(){
    // For each zombe
    for (int z = 0; z < 5; z++){
        // Check it against all the blocks on the screen
        for (int i = 0; i < num_columns; i++){
            for (int j = 0; j < num_rows; j++){
                // If the zombie is standing on a safe block
                if ((blocks[i][j].blocktype == safe || blocks[i][j].blocktype == forbidden)&&(sprites_touch(blocks[i][j].sprite, zombies[z].sprite))){
                    zombies[z].sprite.y = blocks[i][j].sprite.y - zombies[z].sprite.height;            
                    zombies[z].yspeed = 0;

                    // Adjusts zombie speed with the block
                    if (!zombies[z].block_collision){
                    // Initial speeds
                        //zombies[z].xspeed = blocks[i][j].xspeed; // Zombie moves with block speed // POSITIVE WORKS
                        zombies[z].dxspeed = -0.5; // Moves right initially                   
                    }
                    }
